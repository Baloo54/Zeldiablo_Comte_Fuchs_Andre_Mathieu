/**
* méthode décrivant le labyrinthe
* @return la description du labyrinthe 
* en donnant la position du personnage
* le nombre de murs et le nombre de pièges
* le nombre de cases vides
*/
public String descriptionLaby() {
   int nbMurs = 0;
   int nbPieges = 0;
   int nbCasesVides = 0;
   for (int i = 0; i < this.getLength(); i++) {
       for (int j = 0; j < this.getLengthY(); j++) {
           if (this.getMur(i, j) instanceof X) {
               nbMurs++;
           } else if (this.getMur(i, j) instanceof T) {
               nbPieges++;
           } else if (this.getMur(i, j) instanceof CaseVide) {
               nbCasesVides++;
           }
       }
   }
   return "Position du personnage : (" + this.pj.x + "," + this.pj.y + ")\n" +
           "Nombre de murs : " + nbMurs + "\n" +
           "Nombre de pièges : " + nbPieges + "\n" +
           "Nombre de cases vides : " + nbCasesVides;
}
